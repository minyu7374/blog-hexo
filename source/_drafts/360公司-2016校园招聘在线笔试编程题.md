第一题
将考研分数分类

```cpp
#include <cstdio>

int main(){
    int t, a, b, c, d, s;
    scanf("%d", &t);
    while(t--){
        scanf("%d %d %d %d", &a, &b, &c, &d);
        s = a + b + c + d;
        if(a < 60 || b <60 || c < 90 || d < 90 || s < 310)
            printf("Fail\n");
        else if(s < 350)
            printf("Zifei\n");
        else
            printf("Gongfei\n");
    }
    return 0;
}
```
